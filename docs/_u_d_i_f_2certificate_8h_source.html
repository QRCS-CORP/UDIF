694/* Macro alias so existing call sites "just work" without source changes.
695 * The preprocessor selects the right overload based on argument count.
696 * _UDIF_CERT_SER_PICK is a count-based dispatcher. */
697#define _UDIF_CERT_SER_2(out, cert) udif_certificate_serialize_store((out), (cert))
698#define _UDIF_CERT_SER_3(out, len, cert) udif_certificate_serialize((out), (len), (cert))
699#define _UDIF_CERT_SER_PICK(_1,_2,_3,NAME,...) NAME
700
701/* Callers that pass 2 args get serialize2 (returns size_t);
702 * callers that pass 3 args get the original (returns udif_errors).
703 * Usage:
704 * certsz = UDIF_CERT_SERIALIZE(buf, &cert); // 2-arg → size_t
705 * signed_sz = UDIF_CERT_SERIALIZE(buf, sizeof(buf), &c); // 3-arg → udif_errors cast to size_t
706 */
707#define UDIF_CERT_SERIALIZE(...) \
708 _UDIF_CERT_SER_PICK(__VA_ARGS__, _UDIF_CERT_SER_3, _UDIF_CERT_SER_2, _)(__VA_ARGS__)
